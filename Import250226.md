@app.route('/api/athletes/import', methods=['POST'])
def import_athletes():
    if 'file' not in request.files:
        return jsonify({'error': 'No file uploaded'}), 400
    
    file = request.files['file']
    if file.filename == '':
        return jsonify({'error': 'Empty file'}), 400
    
    # Определяем тип файла по расширению
    filename = secure_filename(file.filename)
    ext = filename.rsplit('.', 1)[1].lower() if '.' in filename else ''
    
    imported_count = 0
    errors = []
    
    try:
        if ext == 'csv':
            # Чтение CSV
            stream = io.StringIO(file.stream.read().decode("UTF8"), newline=None)
            csv_reader = csv.DictReader(stream)
            
            for row in csv_reader:
                success, msg = import_athlete_from_dict(row)
                if success:
                    imported_count += 1
                else:
                    errors.append(f"Row {imported_count + 1}: {msg}")
        
        elif ext in ['xlsx', 'xls']:
            # Чтение Excel
            df = pd.read_excel(file)
            for index, row in df.iterrows():
                row_dict = row.to_dict()
                success, msg = import_athlete_from_dict(row_dict)
                if success:
                    imported_count += 1
                else:
                    errors.append(f"Row {index + 2}: {msg}")
        
        else:
            return jsonify({'error': 'Unsupported file format'}), 400
        
        return jsonify({
            'imported': imported_count,
            'errors': errors,
            'total': imported_count + len(errors)
        }), 200
        
    except Exception as e:
        return jsonify({'error': str(e)}), 500

            first_name=first_name,
            last_name=last_name
        ).first()
        
        if existing:
            return False, f"Athlete {first_name} {last_name} already exists"
        
        # Создаём нового спортсмена
        athlete = Athlete(
            first_name=first_name,
            last_name=last_name,
            birth_date=data.get('birth_date') or data.get('birthDate') or data.get('дата_рождения'),
            gender=data.get('gender') or data.get('пол'),
            weight=float(data.get('weight')) if data.get('weight') else None,
            height=float(data.get('height')) if data.get('height') else None,
            country=data.get('country') or data.get('страна'),
            club=data.get('club') or data.get('клуб'),
            coach=data.get('coach') or data.get('тренер'),
            license_number=data.get('license_number') or data.get('license') or data.get('лицензия'),
            ranking=int(data.get('ranking')) if data.get('ranking') else 0,

        )

# API: статистика по турнирам
@app.route('/api/athletes/<int:athlete_id>/statistics', methods=['GET'])
def get_athlete_statistics(athlete_id):
    participations = Participant.query.filter_by(athlete_id=athlete_id).all()
    
    stats = {
        'total_tournaments': len(participations),
        'total_matches': 0,
        'wins': 0,
        'losses': 0,
        'medals': {'gold': 0, 'silver': 0, 'bronze': 0},
        'best_result': None,
        'tournaments': []
    }
    
    for p in participations:
        # Матчи
        matches = Match.query.filter(
            db.or_(Match.participant1_id == p.id, Match.participant2_id == p.id)
        ).all()
        
        tournament_matches = len(matches)
        wins = Match.query.filter_by(winner_id=p.id).count()
        
        stats['total_matches'] += tournament_matches
        stats['wins'] += wins
        stats['losses'] += (tournament_matches - wins)
        
        # Результаты
        result = Result.query.filter_by(participant_id=p.id).first()
        if result:
            stats['tournaments'].append({
                'tournament_id': p.tournament_id,
                'tournament_name': p.tournament.name,
                'position': result.position,
                'medal': result.medal
            })
            
            if result.medal == 'gold':
                stats['medals']['gold'] += 1
            elif result.medal == 'silver':
                stats['medals']['silver'] += 1
            elif result.medal == 'bronze':
                stats['medals']['bronze'] += 1
            
            if not stats['best_result'] or result.position < stats['best_result']['position']:
                stats['best_result'] = {
                    'tournament': p.tournament.name,
                    'position': result.position,
                    'medal': result.medal
                }
    
    return jsonify(stats)

# API: поиск спортсменов (расширенный)
@app.route('/api/athletes/search', methods=['GET'])
def search_athletes():
    query = request.args.get('q', '')
    country = request.args.get('country', '')
    club = request.args.get('club', '')
    min_age = request.args.get('min_age', type=int)
    max_age = request.args.get('max_age', type=int)
    gender = request.args.get('gender', '')
    
    athletes_query = Athlete.query
    
    if query:
        athletes_query = athletes_query.filter(
            db.or_(
                Athlete.first_name.ilike(f'%{query}%'),
                Athlete.last_name.ilike(f'%{query}%'),
                Athlete.country.ilike(f'%{query}%'),
                Athlete.club.ilike(f'%{query}%')
            )
        )
    
    if country:
        athletes_query = athletes_query.filter_by(country=country)
    
    if club:
        athletes_query = athletes_query.filter_by(club=club)
    
    if gender:
        athletes_query = athletes_query.filter_by(gender=gender)
    
    # Фильтр по возрасту (рассчитывается на лету)
    athletes = athletes_query.all()
    if min_age is not None or max_age is not None:
        filtered = []
        for a in athletes:
            age = a.calculate_age()
            if age:
                if min_age is not None and age < min_age:
                    continue
                if max_age is not None and age > max_age:
                    continue
            filtered.append(a)
        athletes = filtered
    
    return jsonify([a.to_dict() for a in athletes])
